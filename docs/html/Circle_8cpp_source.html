33 bool Circle::hit(const Ray& r, float t_min, float t_max, hit_record_t& rec) const {
34  // Vérifier si le rayon intersecte le plan du cercle
35  float denom = dot(normal, r.getDirection());
36  if (fabs(denom) < 1e-6) // Rayon parallèle au plan
37  return false;
38 
39  float t = dot(position - r.getOrigin(), normal) / denom;
40  if (t < t_min || t > t_max)
41  return false;
42 
43  Vec3 hit_point = r.point_at_parameter(t);
44  float distance_squared = (hit_point - position).length_squared();
45  if (distance_squared > radius * radius)
46  return false;
47
