129 void SceneBuilder::createMeshFromObj(std::vector<ObjectProperties> objects)
130 {
131  try {
132  for (const auto& obj : objects) {
133  std::shared_ptr<IMaterial> material = std::make_shared<Texture>(obj.pathTexture);
134  std::ifstream file(obj.pathFile);
135  if (!file.is_open()) {
136  throw Exception("Failed to open file: " + obj.pathFile, Level::LOW);
137  }
138 
139  std::vector<Vec3> vertices;
140  std::vector<std::shared_ptr<IHitable>> triangles;
141  std::string line;
142  std::vector<Vec2> texCoords;
143 
144  while (std::getline(file, line)) {
145  std::stringstream ss(line);
146  std::string type;
147  ss >> type;
148  if (type == "v") {
149  float x, y, z;
150  ss >> x >> y >> z;
151  vertices.emplace_back(x, y, z);
152  } else if (type == "vt") {
153  float u, v;
154  ss >> u >> v;
155  texCoords.emplace_back(u, v);
156  } else if (type == "f") {
157  std::string vertexInfo;
158  std::vector<int> vertexIndices;
159  std::vector<int> textureIndices;
160 
161  while (ss >> vertexInfo) {
162  size_t pos1 = vertexInfo.find('/');
163  size_t pos2 = vertexInfo.find('/', pos1 + 1);
164 
165  int vertexIndex = std::stoi(vertexInfo.substr(0, pos1)) - 1;
166  vertexIndices.push_back(vertexIndex);
167 
168  if (pos1 != std::string::npos && pos1 + 1 != pos2) {
169  int textureIndex = std::stoi(vertexInfo.substr(pos1 + 1, pos2 - pos1 - 1)) - 1;
170  textureIndices.push_back(textureIndex);
171  }
172  }
173  if (vertexIndices.size() >= 3) {
174  for (int i = 0; i < vertexIndices.size() - 2; i++) {
175  triangles.push_back(std::make_shared<Triangle>(
176  vertices[vertexIndices[0]], vertices[vertexIndices[i + 1]], vertices[vertexIndices[i + 2]],
177  texCoords[textureIndices[0]], texCoords[textureIndices[i + 1]], texCoords[textureIndices[i + 2]]
178  ));
179  triangles.back()->setMaterial(material);
180  }
181  }
182  }
183  }
184 
185  Vec3 position(obj.position.x, obj.position.y, obj.position.z);
186  Vec3 rotation(obj.rotation.x, obj.rotation.y, obj.rotation.z);
187  Vec3 scale(obj.scale.x, obj.scale.y, obj.scale.z);
188 
189  file.close();
190  Mesh mesh(triangles, position, rotation, scale);
191  mesh.transformVertices();
192  std::shared_ptr<IHitable> meshPtr = std::make_shared<Mesh>(mesh);
193  scene.addObject(std::move(meshPtr));
194  }
195  } catch (const std::exception& e) {
196  throw Exception("Error creating mesh objects: " + std::string(e.what()), Level::MIDDLE);
197  }
198 }
