43 void Parsing::parsePrimitives(libconfig::Setting& setting, const std::string& type)
44 {
45  for (int i = 0; i < setting.getLength(); ++i) {
46  Primitive p;
47  p.type = type;
48  libconfig::Setting &primitive = setting[i];
49  if (type == "plane") {
50  primitive["first"].lookupValue("x", p.points[0].x);
51  primitive["first"].lookupValue("y", p.points[0].y);
52  primitive["first"].lookupValue("z", p.points[0].z);
53  primitive["second"].lookupValue("x", p.points[1].x);
54  primitive["second"].lookupValue("y", p.points[1].y);
55  primitive["second"].lookupValue("z", p.points[1].z);
56  } else if (type == "sphere") {
57  primitive["center"].lookupValue("x", p.points[0].x);
58  primitive["center"].lookupValue("y", p.points[0].y);
59  primitive["center"].lookupValue("z", p.points[0].z);
60  primitive["radius"].lookupValue("x", p.points[1].x);
61  primitive["radius"].lookupValue("y", p.points[1].y);
62  primitive["radius"].lookupValue("z", p.points[1].z);
63  }
64  libconfig::Setting& mat = primitive["material"];
65  mat.lookupValue("type", p.material.type);
66  mat["vec"].lookupValue("x", p.material.vec.x);
67  mat["vec"].lookupValue("y", p.material.vec.y);
68  mat["vec"].lookupValue("z", p.material.vec.z);
69  if (p.material.type == "metal")
70  mat["fuzz"].lookupValue("x", p.material.fuzz);
71  if (p.material.type == "dielectric")
72  mat["ref_idx"].lookupValue("x", p.material.ref_idx);
73  _primitives.push_back(p);
74  }
