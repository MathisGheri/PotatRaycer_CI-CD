48 void Parsing::parsePrimitives(libconfig::Setting& setting, const std::string& type)
49 {
50  if (setting.getLength() == 0) {
51  Logger *logger = LoggerSingleton::getInstance();
52  logger->log(WARNING, "The section for " + type + " is empty.");
53  return;
54  }
55  for (int i = 0; i < setting.getLength(); ++i) {
56  Primitive p;
57  p.type = type;
58  libconfig::Setting &primitive = setting[i];
59  if (type == "plane") {
60  primitive["first"].lookupValue("x", p.points[0].x);
61  primitive["first"].lookupValue("y", p.points[0].y);
62  primitive["first"].lookupValue("z", p.points[0].z);
63  primitive["second"].lookupValue("x", p.points[1].x);
64  primitive["second"].lookupValue("y", p.points[1].y);
65  primitive["second"].lookupValue("z", p.points[1].z);
66  } else if (type == "sphere") {
67  primitive["center"].lookupValue("x", p.points[0].x);
68  primitive["center"].lookupValue("y", p.points[0].y);
69  primitive["center"].lookupValue("z", p.points[0].z);
70  primitive["radius"].lookupValue("x", p.points[1].x);
71  primitive["radius"].lookupValue("y", p.points[1].y);
72  primitive["radius"].lookupValue("z", p.points[1].z);
73  }
74 
75  libconfig::Setting& mat = primitive["material"];
76  mat.lookupValue("type", p.material.type);
77  mat["vec"].lookupValue("x", p.material.vec.x);
78  mat["vec"].lookupValue("y", p.material.vec.y);
79  mat["vec"].lookupValue("z", p.material.vec.z);
80  if (p.material.type == "metal")
81  mat["fuzz"].lookupValue("x", p.material.fuzz);
82  if (p.material.type == "dielectric")
83  mat["ref_idx"].lookupValue("x", p.material.ref_idx);
84 
85  if (primitive.exists("effect")) {
86  libconfig::Setting &eff = primitive["effect"];
87  eff.lookupValue("type", p.effect.type);
88  if (eff.exists("vec")) {
89  eff["vec"].lookupValue("x", p.effect.color.x);
90  eff["vec"].lookupValue("y", p.effect.color.y);
91  eff["vec"].lookupValue("z", p.effect.color.z);
92  }
93  }
94  _primitives.push_back(p);
95  }
96 }
