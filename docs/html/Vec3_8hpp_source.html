148 inline bool refract(const Vec3& v, const Vec3& n, float ni_over_nt, Vec3& refracted) {
149  Vec3 uv = unit_vector(v);
150  float dt = dot(uv,n);
151  float discriminant = 1.0 - ni_over_nt*ni_over_nt*(1-dt*dt);
152  if(discriminant > 0) {
153  refracted = ni_over_nt*(uv - n*dt) - n*sqrt(discriminant);
154  return true;
155  }
156  return false;
