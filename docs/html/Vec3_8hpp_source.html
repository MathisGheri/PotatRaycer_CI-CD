295 inline bool refract(const Vec3& v, const Vec3& n, float ni_over_nt, Vec3& refracted) {
296  Vec3 uv = unit_vector(v);
297  float dt = dot(uv, n);
298  float discriminant = 1.0 - ni_over_nt * ni_over_nt * (1 - dt * dt);
299  if (discriminant > 0) {
300  refracted = ni_over_nt * (uv - n * dt) - n * sqrt(discriminant);
301  return true;
302  }
303  return false;
304 }
